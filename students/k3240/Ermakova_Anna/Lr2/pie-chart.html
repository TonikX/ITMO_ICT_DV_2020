Источник: Каталог Визуализации Данных

Диаграмма

Описание: Диаграмма показывает у какой расы чаще диагностируют депрессию в процентном соотношении.

Код


    let svg = d3.select("svg"),
        width = svg.attr("width"),
        height = svg.attr("height"),
        radius = Math.min(width, height) / 2;

    let g = svg.append("g")
        .attr("transform", "translate(" + width / 2 + "," + height / 2 + ")");

    let color = d3.scaleOrdinal(['#eac0c0','#b37652','#ef5454','#f5db7b','#7b99f5']);

    let pie = d3.pie().value(function(d) {
        return d.Percent;
    });

    let path = d3.arc()
        .outerRadius(radius)
        .innerRadius(0);

    let label = d3.arc()
        .outerRadius(radius)
        .innerRadius(radius-80);

    d3.csv("depression5.csv", function(error, data) {
        let arc = g.selectAll(".arc")
            .data(pie(data))
            .enter().append("g")

            .attr("class", "arc");
        arc.append("path")
            .attr("d", path)
            .attr("fill", function(d) { return color(d.data.Strata_Name); });

        console.log(arc)

        arc.append("text")
            .attr("transform", function(d) {
                return "translate(" + label.centroid(d) + ")";
            })
            .text(function(d) { return d.data.Strata_Name; });

        const tooltip = svg.append("g").attr("class", "tooltip").style("display", "none");
        tooltip.append("rect").attr("width", 30).attr("height", 20).attr("fill", "white").style("opacity", 0.5);
        tooltip.append("text").attr("x", 15).attr("dy", "1.2em").style("text-anchor", "middle").attr("font-size", "12px").attr("font-weight", "bold");
    });
